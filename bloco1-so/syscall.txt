# PARTE B — Programa C com write() direto

# Cole aqui o código de syscall_direto.c que você escreveu:

#include <unistd.h>

int main() {
    const char msg[] = "Modo usuario -> kernel via write()\n";
    write(1, msg, sizeof(msg) - 1);
    return 0;
}
